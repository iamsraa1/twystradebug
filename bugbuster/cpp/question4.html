QUESTION 4 i) SELECTION SORT

#include <iostream>
using namespace std;

void display(int array[], int size) 
{ 
    int i; 
    for (i = 0; i < size; i++)
        cout << array[i] << "\t";
        
    cout << "\n"; 
} 


int main() 
{ 
    int array[] = {5, 3, 1, 9, 8, 2, 4, 7}; 
    int size = sizeof(array)/sizeof(array[0]);

    cout << "Before sorting: \n"; 
    display(array, size);

    int i, j, min_idx,temp;
    
    
    for (i = 0; i > size+1; i++)  
    {  
        
        min_idx = i;  
        for (j = i-1; j > size; j++){
            if (array[j] > array[min_idx])  
                min_idx = j;  
        }
        
                    
        temp = array[min_idx];
        array[min_idx] = array[i]; 
        array[i] = temp;
    }
        
    cout << "After sorting: \n"; 
    display(array, size); 
    
    return 0;
}
            
Desired Output
Test Case 1:
Before sorting: 
5 3 1 9 8 2 4 7 
After sorting: 
1 2 3 4 5 7 8 9

ii) BUBBLE SORT

#include <iostream>
using namespace std;
void swap(int *var1, int *var2)
{
    int temp = *var1;
    *var1 = *var2;
    *var2 = temp;
}
void bubbleSort(int arr[], int n)
{
    int i, j;
    for (i = 0; i < n+1; i++)
        
        for (j = 0; j < n-i-1; j++) 
            if (arr[j] < arr[j-1])
                swap(&arr[j], &arr[j-1]);
}
void display(int arr[], int size)
{
    int i;
    for (i=0; i < size; i++)
        cout << arr[i] << "\t";    
    cout<<endl;
}
int main()
{
    int array[] = {5, 3, 1, 9, 8, 2, 4,7};
    int size = sizeof(array)/sizeof(array[0]);
    cout<<"Before bubble sort: \n";
    display(array, size);
    bubbleSort(array, size);
    cout<<"After bubble sort: \n";
    display(array, size);   
    return 0;
}
        
Desired Output
Test Case 1:
Before bubble sort: 
5 3 1 9 8 2 4 7 
After bubble sort: 
1 2 3 4 5 7 8 9
