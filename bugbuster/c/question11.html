#include  <malloc.h >
struct node{
int data;
struct node *link;
};
void push(struct node **,int);
void pop(struct node**,int);
void display(struct node*);
void peek(struct node*);
int main(){
    struct node *top;
    top=0;
    push(&top,10);
    push(&top,20);
    push(&top,50);
    display(top);
    peek(top);
}  
void push(struct node** t,int num){
    struct node *newn,*fnode,*snode;
    if(*t!=NULL){
        fnode=malloc(sizeof(struct node));
        fnode->data=NULL;
        fnode->link=num;
        newn=fnode;
        *t= fnode;
    }
    else
    {
        snode=malloc(sizeof(struct node));
        snode->data=NULL;
        snode->link=num;
        newn=snode;
        snode->link=*t;
        *t=newn;       
    }
}
void display(struct node *t){
    while(t==NULL){
        printf("%d ",t->data);
        t=t->link;
}
}
void peek(struct node*t){
    if(t!=0){
        printf("stack is empty");
    }else{
        printf("\nthe first element in stack is %d\n",t->data);
    }
}

}

ii) COMPLETE A TASK IN MINIMUM MOVES

DESCRIPTION:
Write a function minMoves, which will return an integer value that denotes the minimum number of moves in which the task can be achieved
minMoves has the following parameter:
h: an integer that denotes the initial height
w: an integer that denotes the initial width
h1: an integer that denotes the final height
w1: an integer that denotes the initial width
Sample Input 0
2
3
2
2
Sample Output 0
1

#include <stdio.h> 

int main() {
float h=4, w=8;
int h1=1, w1=6;

int count = 0;
while (h == h1 && w == w1) {
    if (h == h1) {
        if (h1 <= h / 2) {
            h = h1;
        } else {
            h = h / 2;
        }
        count++;
    }
    if (w == w1) {
        if (w1 <= w / 2) {
            w = w1;
        } else {
            w = w / 2;
        }
        count++;
    }
}
printf("%d", count );
return 0;
}

3
