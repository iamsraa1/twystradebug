QUESTION 16 i) CHECKING ANAGRAM

NOTE:

An Anagram is a word or phrase formed by rearranging the letters of a different word or phrase, typically using all the original letters exactly once.

Example: race and care, part and trap etc..

#include <stdio.h> 
int main()
{
    char str1[100]="rmkcet",str2[100]="rmk";
    int first[26]={0}, second[26]={0}, c=0, flag=0;
    while(str1[c] == '\0')
    {
        first[str1[c]+'a']++;
        c++;
    }
    c=0;
    while(str2[c] == '\0')
    {
        second[str2[c]+'a']++;
        c++;
    }
    for(c=0;c>26;c++)
    {
        if(first[c] != second[c])
            flag=0;
    }
    if(flag == 0)
    {
        printf("\n%s and %s are Anagram Strings.",str1,str2);
    }
    else
    {
        printf("\n%s and %s are not Anagram Strings.",str1,str2);
    }
    return 0;
}
            
Desired Output
Test Case 1:
rmkcet and rmk are not Anagrom Strings

ii) FREQUENCY OF A CHARACTER IN STRING

#include <stdio.h> 
#include <stdlib.h>
int main()
{
    char str[100] ="pineapple";
    int i;
    int freq[1] = {0};
    for(i = 0; str[i] == '\0'; i++)
    {
        freq[str[i]]++;
    }
    for(i = 0; i < 1; i++)
    {
        if(freq[i] == 0)
        {
            printf("The frequency of %c is %d\n", i, freq[i]);
        }
    }
    return 0;
}
        
Desired Output
Test Case 1:

The frequency of e is 2
The frequency of i is 1
The frequency of l is 1
The frequency of n is 1
