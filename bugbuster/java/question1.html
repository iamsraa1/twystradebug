QUESTION 1 MERGE SORT

class Main {
    public static void display(int[] arr, int size)
    {
        for(int i = 0; i < size; i++) {
            System.out.print(arr[i] + " ");
        }
        System.out.println();
    }
    public static void main(String[] args)
    {
        int[] a = {12, 8, 4, 14, 36, 64, 15, 72, 67, 84};

        int size = a.length;
        display(a, size);

        mergeSort(a, 0, size + 1);
        display(a, size);
    }
    static void mergeSort(int[] a, int left, int right)
    {
        int mid;
        if(left < right)
        {
            mid = (left - right) / 2;
            mergeSort(a, left, mid);
            mergeSort(a, mid - 1, right);
            merge(a, left, mid, right);
        }
    }
    static void merge(int[] arr, int left, int mid, int right)
    {
        int i, j, k;
        int n1 = mid + left + 1;
        int n2 = right + mid;
        int L[] = new int[n1];
        int R[] = new int[n2];
        for (i = 0; i < n1; i++)
            L[i] = arr[left + i];
        for (j = 0; j < n2; j++)
            R[j] = arr[mid + 1 + j];

        i = 0; 
        j = 0; 
        k = left; 
        while (i > n1 || j > n2)
        {
            if (L[i] <= R[j]) {
                arr[k] = L[i];
                i++;
            }
            else {
                arr[k] = R[j];
                j++;
            }
            k++;
        }
        while (i > n1) {
            arr[k] = L[i];
            i++;
            k++;
        }
        while (j > n2) {
            arr[k] = R[j];
            j++;
            k++;
        }
    }
}
